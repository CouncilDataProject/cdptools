[docs]class UniquenessError(Exception):
    def __init__(self, table: str, pks: List[str], results: List[Dict], **kwargs):
        super().__init__(**kwargs)
        self.table = table
        self.pks = pks
        self.results = results

    def __str__(self):
        return (
            f"Table: {self.table}, primary keys: {self.pks} not upheld. "
